/** APPS SCRIPT CODE (paste ke Code.gs) **/ function doGet(e){ const sheet = SpreadsheetApp.getActive().getSheetByName("Data"); let data = sheet.getDataRange().getValues(); data.shift(); if (e.parameter.read) { let output = data.map(r => ({ nama:r[0], nip:r[1], unit:r[2], jabatan:r[3], ttd:r[4] })); return ContentService.createTextOutput(JSON.stringify(output)); } if (e.parameter.excel) return exportExcel(); if (e.parameter.pdf) return exportPDF(); } function doPost(e){ const sheet = SpreadsheetApp.getActive().getSheetByName("Data"); let obj = JSON.parse(e.postData.contents); sheet.appendRow([obj.nama, obj.nip, obj.unit, obj.jabatan, obj.ttd]); return ContentService.createTextOutput(JSON.stringify({status:"OK"})); } function exportExcel(){ const ss = SpreadsheetApp.getActive(); const blob = ss.getBlob().setName("Absensi.xlsx"); return ContentService.createBinaryOutput(blob.getBytes()) .setMimeType("application/vnd.openxmlformats-officedocument.spreadsheetml.sheet"); } function exportPDF(){ const ss = SpreadsheetApp.getActive(); const pdf = ss.getAs("application/pdf").setName("Absensi.pdf"); return ContentService .createBinaryOutput(pdf.getBytes()) .setMimeType("application/pdf"); }
